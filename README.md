# staypay_iot

=============================
TO START OPERATIONS
=============================
run start.sh

  sh start.sh

=============================
TO UNLOCK DOOR
=============================
send arguments 0 an 1

 sh run.sh 0 1
 
=============================
TO TURN ON NIGHT LIGHT
=============================
send arguments 1 an 1

 sh run.sh 1 1
 
=============================
TO TURN OFF NIGHT LIGHT
=============================
send arguments 1 an 0

 sh run.sh 1 0
 
 =============================
TO TURN ON LIGHT
=============================
send arguments 2 an 1

 sh run.sh 2 1

=============================
TO TURN OFF LIGHT
=============================
send arguments 2 an 0

 sh run.sh 2 0
 
 =============================
TO TURN ON AC
=============================
send arguments 3 an 1

 sh run.sh 3 1

=============================

 sh run.sh 3 0
